const REGRAS = [/* order-sensitive default rules */];

export function categorizar(desc) {
  const upper = normalizarTexto(desc);
  for (const regra of REGRAS) {
    if (regra.keywords.some(kw => upper.includes(normalizarTexto(kw)))) {
      return regra.cat;
    }
  }
  return 'Outros';
}

From js/utils/full-backup-io.js:
```javascript
export function validateFullBackupPayload(payload) {
  ensureObject(payload, 'JSON invalido.');
  if (payload.versao !== FULL_BACKUP_VERSION) throw new Error(...);
  if (payload.tipo !== FULL_BACKUP_TYPE) throw new Error(...);

  const stores = {};
  for (const storeName of FULL_BACKUP_STORE_NAMES) {
    if (!Array.isArray(payload.stores[storeName])) {
      throw new Error(`Backup invalido: store "${storeName}" ausente ou malformada.`);
    }
    stores[storeName] = cloneValue(payload.stores[storeName]);
  }
  return { versao: FULL_BACKUP_VERSION, tipo: FULL_BACKUP_TYPE, exportadoEm: payload.exportadoEm, dbName: payload.dbName, stores };
}
```
